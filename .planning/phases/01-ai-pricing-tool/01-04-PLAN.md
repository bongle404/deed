    Add JS functions to the existing script block. Place them near the existing form logic:

    ```javascript
    let estimateDebounceTimer;
    const AU_CURRENCY = new Intl.NumberFormat('en-AU', { style: 'currency', currency: 'AUD', maximumFractionDigits: 0 });

    function formatSoldDate(dateStr) {
      if (!dateStr) return '—';
      const d = new Date(dateStr);
      return d.toLocaleDateString('en-AU', { month: 'short', year: 'numeric' });
    }

    function renderEstimateWidget(data) {
      document.getElementById('estimate-loading').style.display = 'none';
      document.getElementById('estimate-error').style.display = 'none';
      document.getElementById('estimate-empty').style.display = 'none';

      if (!data || data.error) {
        document.getElementById('estimate-error').style.display = 'block';
        return;
      }

      document.getElementById('est-low').textContent = data.low ? AU_CURRENCY.format(data.low) : '—';
      document.getElementById('est-mid').textContent = data.mid ? AU_CURRENCY.format(data.mid) : '—';
      document.getElementById('est-high').textContent = data.high ? AU_CURRENCY.format(data.high) : '—';

      const confidenceLabels = { HIGH: 'High confidence — 5+ comparable sales', MEDIUM: 'Medium confidence — 3–4 comparable sales', LOW: 'Low confidence — fewer than 3 comparable sales' };
      document.getElementById('est-confidence').textContent = confidenceLabels[data.confidence] || '';
      document.getElementById('est-data-note').textContent = `Based on ${data.comp_count} comparable sale${data.comp_count !== 1 ? 's' : ''} from Proptech Data`;

      const tbody = document.getElementById('comps-tbody');
      tbody.innerHTML = '';
      (data.comparables || []).slice(0, 8).forEach(comp => {
        const tr = document.createElement('tr');
        tr.innerHTML = `
          <td>${comp.address || '—'}</td>
          <td class="comp-price-cell">${AU_CURRENCY.format(comp.price)}</td>
          <td>${comp.bedrooms || '—'}/${comp.bathrooms || '—'}</td>
          <td>${formatSoldDate(comp.sold_date)}</td>
        `;
        tbody.appendChild(tr);
      });

      document.getElementById('estimate-content').style.display = 'block';
    }

    async function refreshEstimate() {
      // Get current form values — use the actual field IDs from sell.html
      const suburb = (document.getElementById('suburb') || {}).value || '';
      const bedrooms = (document.getElementById('bedrooms') || {}).value || '';
      const bathrooms = (document.getElementById('bathrooms') || {}).value || '';
      const propertyType = (document.getElementById('property_type') || {}).value || 'house';

      if (!suburb || !bedrooms) {
        document.getElementById('estimate-empty').style.display = 'block';
        document.getElementById('estimate-content').style.display = 'none';
        return;
      }

      document.getElementById('estimate-loading').style.display = 'block';
      document.getElementById('estimate-content').style.display = 'none';
      document.getElementById('estimate-empty').style.display = 'none';

      try {
        const res = await fetch('/api/price-estimate', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ suburb, bedrooms: parseInt(bedrooms), bathrooms: parseInt(bathrooms), property_type: propertyType })
        });
        const data = await res.json();
        renderEstimateWidget(data);
      } catch (err) {
        document.getElementById('estimate-loading').style.display = 'none';
        document.getElementById('estimate-error').style.display = 'block';
      }
    }

    // Attach debounced listeners — find the actual field IDs used in sell.html and use them
    ['bedrooms', 'bathrooms', 'land_size'].forEach(fieldId => {
      const el = document.getElementById(fieldId);
      if (el) {
        el.addEventListener('input', () => {
          clearTimeout(estimateDebounceTimer);
          estimateDebounceTimer = setTimeout(refreshEstimate, 600);
        });
      }
    });
    // Also trigger on suburb change (not debounced — suburb changes are discrete)
    const suburbEl = document.getElementById('suburb');
    if (suburbEl) suburbEl.addEventListener('change', refreshEstimate);
    ```
