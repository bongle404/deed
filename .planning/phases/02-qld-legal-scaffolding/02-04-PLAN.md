**JavaScript renderChecklist() function (add to both pages' scripts):**
```javascript
function renderChecklist(items, containerId) {
  const container = document.getElementById(containerId);
  if (!container) return;
  const icons = { pass: '✓', fail: '✕', warning: '⚠', info: 'ⓘ' };
  const colours = { pass: '#16a34a', fail: '#dc2626', warning: '#d97706', info: '#2563eb' };
  container.innerHTML = items.map(item => `
    <div class="checklist-item">
      <span class="checklist-icon" style="color:${colours[item.status]}">${icons[item.status]}</span>
      <div>
        <div class="checklist-label">${item.label}</div>
        ${item.note ? `<div class="checklist-note">${item.note}</div>` : ''}
      </div>
    </div>
  `).join('') + '<p class="checklist-self-reported">Certificates are self-reported by the seller. Buyers should verify independently before exchanging contracts.</p>';
}
```
